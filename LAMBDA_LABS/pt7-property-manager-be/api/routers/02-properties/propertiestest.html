const request = require("supertest"); const propertiesRouter = require("./properties-router"); describe("Properties Router /", () => { it("should return 200", () => { request(propertiesRouter) .get("/") .then(res => { expect(res.status).toBe(200); }); }); }); describe("Get all properties by manager id /:id", () => { it("should return 200 ", () => { request(propertiesRouter) .get("/:id") .then(res => { expect(res.status).toBe(200); }); }); }); describe("Edits property by ID", () => { it("should return 200", () => { request(propertiesRouter) .put("/:id") .then(res => { expect(res.status).toBe(200); }); }); }); describe("Get all properties by manager id", () => { it("should return 200", () => { request(propertiesRouter) .put("/manager/:id") .then(res => { expect(res.status).toBe(200); }); }); }); describe("Deletes property by ID", () => { it("should return 200", () => { request(propertiesRouter) .delete("/:id") .then(res => { expect(res.status).toBe(200); }); }); });
